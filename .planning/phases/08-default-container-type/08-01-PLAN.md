    In lib/menu.sh, replace menu_select_container_type (lines 121-134) with:

    ```bash
    menu_select_container_type() {
        local default_type
        default_type="$(settings_get '.defaults.container_type')"

        if [[ -n "$default_type" ]]; then
            # User has set a preference — auto-skip
            local label
            if [[ "$default_type" == "gui" ]]; then label="GUI Desktop"; else label="Minimal CLI"; fi
            echo "-> Using default: ${label}" >&2
            echo "  Change in [P] Preferences" >&2
            echo "$default_type"
            return
        fi

        # Null/unset — show interactive prompt
        echo ""
        echo "Select container type:"
        echo "  [1] Minimal CLI"
        echo "  [2] GUI Desktop"
        echo ""
        local choice
        read -rp "Type [1]: " choice
        case "${choice:-1}" in
            1) echo "cli" ;;
            2) echo "gui" ;;
            *) msg_error "Invalid selection. Using Minimal CLI."; echo "cli" ;;
        esac
    }
    ```

    Key detail: Auto-skip message goes to stderr (>&2) so command substitution in menu_action_new only captures the type string. This avoids Pitfall 3 from RESEARCH.md. Use plain ASCII arrow "->", not unicode.
